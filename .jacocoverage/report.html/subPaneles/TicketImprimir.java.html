    /**
     *
     * @param graf
     * @param pagefor
     * @param Index
     * @return
     */
    public int print(Graphics graf, PageFormat pagefor, int Index){
        if (Index>0){
            return NO_SUCH_PAGE;
        }
        Graphics2D ar=(Graphics2D) graf;
        ar.translate(pagefor.getImageableX()*20, pagefor.getImageableY()*40);
        ar.scale(2.0, 1.5);
        Panel_Ticket.printAll(graf);
        
        
        return PAGE_EXISTS;
    }
